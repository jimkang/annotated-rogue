    int
get_rand(int x, int y)
{
  • ¶ 
    	int r, t;
	long lr;
  • ¶ 
    	if (x > y) {
		t = y;
		y = x;
		x = t;
	}
  • ¶ 
    	lr = rrandom();
	lr &= (long) 0x00003fff;
	r = (int)lr;
	r = (r % ((y - x) + 1)) + x;
	return(r);
}
